def display_table(gen_cell, n_rows=2, n_cols=2):
    html = ''
    html += '<table>'
    for row_id in range(n_rows):
        html += '<tr>'
        for col_id in range(n_cols):
            html += '<td>' + gen_cell(row_id, col_id) + '</td>'
        html += '</tr>'
    html += '</table>'
    display(HTML(html))

names = ['One', 'Done', 'Won', 'Bet', 'Jet', 'Threat']
# scores = [np.random.randn(8) for i in range(len(names))]
scores = [np.load('%d-score.npy' % (i + 1)) for i in range(len(names))]

def gen_cell(row_id, col_id):
    if row_id == 0 and col_id == 0:
        return ''
    elif row_id == 0:
        return names[col_id - 1]
    elif col_id == 0:
        return names[row_id - 1]
    else:
        return '%f' % np.sqrt(np.sum((scores[row_id - 1] - scores[col_id - 1])**2))
