SAMPLE_RATE = 8000

def gen_sine(freq_hz, n_secs, sample_rate):
    n_timesteps = int(n_secs * sample_rate)
    ts = np.linspace(0, n_secs, num=n_timesteps)
    x = np.sin(2 * np.pi * ts * freq_hz)
    return x
