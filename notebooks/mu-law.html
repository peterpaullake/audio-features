def plot_spec(ax, x):
    ax.specgram(x, Fs=SR, interpolation='nearest')
    ax.set_xlabel('Time (secs)')
    ax.set_ylabel('Frequency (Hz)')
    
natural_period = 2 * (C - 1)
n_samples = 4 * natural_period

for period_multiplier in [1]:
    period = period_multiplier * natural_period
    freq = SR / period
    sample_ids = np.arange(n_samples)
    t = sample_ids / SR
    
    x_pure = np.sin(2 * np.pi * freq * t)
    x_indices = wf.make_sine(C, n_samples, period_multiplier, phi=0)
    x_encoded = 2 * x_indices / (C - 1) - 1
    x_decoded = wf.audio_data.mu_law_expansion(x_encoded, C)

    fig, axes = plt.subplots(3, dpi=150, figsize=(10, 6))
    fig.suptitle('%.2f Hz' % freq)
    axes[0].set_title('Pure sine')
    axes[1].set_title('After mu-law encoding and quantization')
    axes[2].set_title('After un-quantizing and mu-law decoding')
    
    axes[0].plot(t, x_pure)
    axes[1].plot(t, x_encoded, '.', markersize=1)
    axes[2].plot(t, x_decoded)
    
    for i, ax in enumerate(axes):
        if i == len(axes) - 1:
            ax.set_xlabel('Time (secs)')
        else:
            ax.set_xticks([])
    
    plt.tight_layout()
    plt.show()
