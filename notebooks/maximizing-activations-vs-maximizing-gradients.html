def sigmoid(x):
    return 1 / (1 + np.exp(-x))

def demo_model(n_examples=4):
    model = lsc.load_model()
    imgs, angles = lsc.gen_random_examples(4)
    
    predicted_angles = model(torch.Tensor(imgs)).detach().numpy()
    
    for i, (img, angle, predicted_angle) in enumerate(zip(imgs, angles, predicted_angles)):
        fig, ax = plt.subplots()
        if i == 0:
            fig.suptitle('Line segment model', y=1.1, size=20)
        deg = lambda x : x / np.pi * 180
        ax.set_title('Actual angle = %f degrees\nPredicted angle = %f degrees' % (deg(angle), deg(predicted_angle)))
        ax.imshow(img, origin='lower')
        plt.show()
