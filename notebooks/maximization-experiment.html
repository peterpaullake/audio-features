default_props = {'init_type': 'random',
                 'conv_id': 0,
                 'channel_id': 0,
                 'output_type': 'res',
                 'activation_type': 'mean2',
                 'f0': 0,
                 'f1': 60,
                 'n_frames': 20,
                 'unroll_every': 100}

def config_freq_axis(ax, num_bins):
    fmin, fmax = 125, 7600
    mel_to_f = lambda mel : 700 * (np.exp(mel / 1127) - 1)
    f_to_mel = lambda f : 1127 * np.log(1 + f / 700)
    lower_mel = f_to_mel(fmin)
    upper_mel = f_to_mel(fmax)
    
    def yformat(value, pos):
        # value = 1 + value # num_bins - 1 - value
        step = (upper_mel - lower_mel) / num_bins
        mel = lower_mel + value * step
        return '%.2f' % mel_to_f(mel)
    ax.yaxis.set_major_formatter(plt.FuncFormatter(yformat))
    ax.set_ylabel('Frequency (Hz)')

def plot_mel(ax, mel, draw_title=True, aspect='auto'):
    if draw_title:
        ax.set_title('Optimized mel spectrogram local conditioning features')
    config_freq_axis(ax, 80)
    ax.get_xaxis().set_visible(False)
    # ax.get_yaxis().set_visible(False)
    ax.imshow(mel.transpose(), aspect=aspect, cmap='gray', origin='lower', interpolation='nearest')

def plot_wave(ax, wave, t0, t1):
    def format_func(value, tick_number):
        return '%.2f' % (value / RATE)
    
    ax.set_title('Unrolled audio')
    ax.set_xlabel('Time (seconds)')
    ax.xaxis.set_major_formatter(plt.FuncFormatter(format_func))
    ax.plot(wave)
    ax.axvline(x=t0, linestyle='--', color='red', label=r'$t_0$')
    ax.axvline(x=t1, linestyle='--', color='black', label=r'$t_1$')
    ax.legend()

def plot_activations(ax, activations):
    ax.set_title('Loss curve')
    ax.set_xlabel('Steps')
    ax.set_ylabel('Activation')
    ax.plot(activations)

def gen_title(props):
    s = ''
    s += 'Maximizing '
    if props['activation_type'] == 'mean':
        s += 'mean of activations '
    if props['activation_type'] == 'mean2':
        s += 'mean of squared activations '
    elif props['activation_type'] == 'band':
        s += 'frequencies from %d to %d Hz of activations ' % (props['f0'], props['f1'])
    s += '\nof layer %d, ' % props['conv_id']
    s += 'channel %d ' % props['channel_id']
    s += r'from $t_0$ to $t_1$'
    return s

def show_entry(entry):
    mel = entry['mel']
    wave = entry['wave']
    activations = entry['activations']
    dpi = 100
    figsize = (6, 6)
    
    fig, axes = plt.subplots(3, dpi=dpi, figsize=figsize)
    plot_mel(axes[0], mel)
    plot_wave(axes[1], wave, len(wave) // 3, 2 * len(wave) // 3)
    plot_activations(axes[2], activations)
    
    try:
        props = entry['props']
    except KeyError:
        props = default_props
    title = gen_title(props)
    
    fig.suptitle(title, fontsize=16, y=1.1)
    plt.tight_layout()
    plt.show()
    
    fig, ax = plt.subplots(figsize=(6, 6))
    plot_mel(ax, mel, False, 'equal')
    plt.show()
    
    display(Audio(wave, rate=22050))

for i in [2, 0, 1]:
    show_entry(pickle.load(open('maximization-experiment/%d.pickle' % i, 'rb'))['data'])
