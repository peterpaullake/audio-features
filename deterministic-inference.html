def display_wave(wave, title=None):
    fig, ax = plt.subplots(figsize=(10, 2), dpi=DPI)
    fig.suptitle(title)
    ax.plot(wave)
    config_time_axis(ax, len(wave), len(wave) / RATE)
    plt.show()
    display(Audio(wave, rate=RATE))
